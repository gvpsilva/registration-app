# registration-app

#Instalação do Node.js - (Linux-WSL)
curl -fsSL https://deb.nodesource.com/setup_lts.x | sudo -E bash -
sudo apt-get install -y nodejs

#Configuração do Projeto
mkdir registration-app
cd registration-app
npm init -y

#Instalação de Dependências
npm install express
npm install

#Iniciar Servidor
npm start
